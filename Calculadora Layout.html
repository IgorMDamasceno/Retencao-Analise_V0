    function calcFromInputs(investimento, cpa, roasP, aprovLeadsP, aprovCicloP, rpsUsd){

      const leads = (cpa > 0) ? (investimento / cpa) : 0;

      const valorRec = investimento * (roasP/100);

      const chegada = leads * (aprovLeadsP/100);

      const sessoes = chegada * (aprovCicloP/100);

      const receitaCiclo = (sessoes * rpsUsd / 1000) * USD_BRL_RATE;

      const valorFinal = valorRec + receitaCiclo;

      const percentFinal = (investimento > 0) ? (valorFinal / investimento) * 100 : 0;

      return { leads, valorRec, chegada, sessoes, receitaCiclo, valorFinal, percentFinal };

    }

      const aprovCicloP = n('op_aprovCiclo');

      const rpsUsd = n('op_rps');

    }

    function computeAverages(list){

      const sum = list.reduce((acc, op) => {

        acc.investimento += op.investimento || 0;

        acc.cpa += op.cpa || 0;

        acc.roas += op.roasPercent || 0;

        acc.aprovLeads += op.aproveitamentoLeadsPercent || 0;

        acc.aprovCiclo += op.aproveitamentoCicloInicialPercent || 0;

        acc.rps += op.rpsCicloInicialUsd || 0;

        return acc;

      }, { investimento: 0, cpa: 0, roas: 0, aprovLeads: 0, aprovCiclo: 0, rps: 0 });

      const count = list.length || 1;

      return {

        investimento: sum.investimento / count,

        cpa: sum.cpa / count,

        roas: sum.roas / count,

        aprovLeads: sum.aprovLeads / count,

        aprovCiclo: sum.aprovCiclo / count,

        rps: sum.rps / count

      };

    }

    function valueSizeClass(text, variant='main'){

      const len = String(text ?? '').length;

      if (variant === 'mid') {

        if (len > 16) return 'text-xs';

        if (len > 13) return 'text-sm';

        if (len > 10) return 'text-base';

        return 'text-lg';

      }

      // default (main/final)

      if (len > 18) return 'text-xs';

      if (len > 15) return 'text-sm';

      if (len > 11) return 'text-base';

      return 'text-lg';

    }


    function buildFunnelHorizontalHtml(op, avgMetrics=null){

      const c = calcFromInputs(

        op.investimento, op.cpa, op.roasPercent,

        op.aproveitamentoLeadsPercent, op.aproveitamentoCicloInicialPercent,

        op.rpsCicloInicialUsd

      );


      const percentBar = clamp(c.percentFinal, 0, 120);


      const compareInvest = avgMetrics ? percentDelta(op.investimento, avgMetrics.investimento) : null;

      const compareRoas = avgMetrics ? percentDelta(op.roasPercent, avgMetrics.roas) : null;

            </div>

          </div>

            <div class="text-lg font-extrabold text-white">${brl(c.valorFinal)}</div>

            <div class="text-[11px] text-slate-400">${pct(c.percentFinal)} do investimento</div>

      `;

    function percentDelta(value, avg){

      if (!avg) return { text: '0,00%', isPositive: true };

      const diffPct = ((value - avg) / avg) * 100;

      const isPositive = diffPct >= 0;

      const text = `${isPositive ? '+' : ''}${diffPct.toLocaleString('pt-BR', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}%`;

      return { text, isPositive };

    }


    function nodeHtml(topLabel, title, value, lines, isPlain=false, compare=null){

      const valueClass = valueSizeClass(value, 'main');

      const cardClass = isPlain ? 'funnel-node plain' : 'funnel-node funnel-card';

      const paddingClass = isPlain ? 'p-0' : 'p-4';

      return `

        <div class="${cardClass} rounded-2xl ${paddingClass}">

          ${topLabel ? `<div class="text-xs font-bold text-slate-300">${escapeHtml(topLabel)}</div>` : ''}

          <div class="mt-1 text-sm font-semibold text-white">${escapeHtml(title)}</div>

          <div class="mt-2 ${valueClass} leading-tight font-extrabold text-indigo-200 overflow-hidden text-ellipsis whitespace-nowrap">${escapeHtml(value)}</div>

          <div class="mt-3 space-y-1">

            ${lines.map(([k,v]) => `<div class="flex items-center gap-2 text-[11px] text-slate-400 min-w-0"><span class="min-w-0">${escapeHtml(k)}</span><span class="ml-auto text-slate-200 font-semibold text-right whitespace-nowrap overflow-hidden text-ellipsis max-w-full">${escapeHtml(v)}</span></div>`).join('')}

          </div>

          ${compare ? `<div class="compare-foot ${compare.isPositive ? 'positive' : 'negative'}">Δ ${compare.text}</div>` : ''}

        </div>

      `;

    }


    function finalHtml(topLabel, title, value, lines){

      const valueClass = valueSizeClass(value, 'main');

      return `

        <div class="funnel-node final rounded-2xl p-4 funnel-card">

          <div class="text-xs font-bold text-indigo-200 text-center">${escapeHtml(topLabel)}</div>

          ${title ? `<div class="mt-1 text-sm font-semibold text-white">${escapeHtml(title)}</div>` : ''}

          <div class="mt-2 ${valueClass} leading-tight font-extrabold text-white text-center">${escapeHtml(value)}</div>

          <div class="mt-3 space-y-1">

            ${lines.map(([k,v]) => `<div class="flex items-center gap-2 text-[11px] text-indigo-100/80 min-w-0"><span class="min-w-0">${escapeHtml(k)}</span><span class="ml-auto text-white font-semibold text-right whitespace-nowrap overflow-hidden text-ellipsis max-w-full">${escapeHtml(v)}</span></div>`).join('')}

          </div>

        </div>

      `;

    }


    function groupHtml(label, leftHtml, rightHtml, variant=''){

      const groupClass = variant ? `funnel-group ${variant}` : 'funnel-group';

      return `

        <div class="${groupClass}">

          <div class="funnel-group__header">${escapeHtml(label)}</div>

          <div class="funnel-group__body">

            ${leftHtml}

            ${arrowHtml(true)}

            ${rightHtml}

          </div>

        </div>

      `;

    }


    function midHtml(prefix, title, bigValue, chips, compare=null){

      const valueClass = valueSizeClass(bigValue, 'mid');

      return `

        <div class="funnel-mid">

          ${prefix ? `<div class="text-[11px] text-slate-400">${escapeHtml(prefix)}</div>` : ''}

          <div class="text-sm font-bold text-white">${escapeHtml(title)}</div>

          <div class="mt-1 ${valueClass} leading-tight font-extrabold text-indigo-200">${escapeHtml(bigValue)}</div>

          <div class="mt-3 flex flex-col items-center gap-2">

            ${chips.map(([k,v]) => `<div class="chip"><span>${escapeHtml(k)}</span><b class="text-slate-100">${escapeHtml(v)}</b></div>`).join('')}

          </div>

          ${compare ? `<div class="compare-foot ${compare.isPositive ? 'positive' : 'negative'}">Δ ${compare.text}</div>` : ''}

        </div>

      `;

    }


    function arrowHtml(compact=false){

      const cls = compact ? 'funnel-arrow compact' : 'funnel-arrow';

      return `

        <div class="${cls}" aria-hidden="true">

          <svg width="22" height="22" viewBox="0 0 24 24" fill="none">

            <path d="M5 12h14" stroke="#a5b4fc" stroke-width="2" stroke-linecap="round"/>

            <path d="M13 6l6 6-6 6" stroke="#a5b4fc" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"/>

          </svg>

        </div>

      `;

    }


    function escapeHtml(s){

      return String(s ?? '')

        .replaceAll('&','&amp;')

        .replaceAll('<','&lt;')

        .replaceAll('>','&gt;')

        .replaceAll('"','&quot;')

        .replaceAll("'","&#039;");

    }
